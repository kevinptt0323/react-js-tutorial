JSX

Write HTML inside javascript


let username = 'kevinptt';
ReactDOM.render((
  <div>
    <h1>Hello, {username}!</h1>
  </div>
), document.getElementById('app'));

without JSX


let username = 'kevinptt';
ReactDOM.render(
  React.createElement('div', null, 
    React.createElement('h1', null, `Hello, ${username}!`)
  ),
  document.getElementById('app')
);

write javascript inside "{ }"


let checked = true;
ReactDOM.render((
  <h1 className={checked ? 'show' : 'hide'}>Hello, world!</h1>
), document.getElementById('app'));
				

var arr = [
  <h1>Hello world!</h1>,
  <h2>React is awesome</h2>,
];

  <div>{arr}</div>,
  document.getElementById('app')
);
